-- Codigo para generar las tablas de la DB
CREATE TABLE marcas(
marcas_id int AUTO_INCREMENT PRIMARY KEY, 
nombre_marca VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE modelos(
 marcas_id int,
 modelos_id int PRIMARY KEY AUTO_INCREMENT,
 modelo_ref VARCHAR(50) NOT NULL,
 -- color_01 VARCHAR(30), SE ELIMINO
 -- color_02 VARCHAR(30), SE ELIMINO
 -- color_03 VARCHAR(30), SE ELIMINO
 FOREIGN KEY (marcas_id) REFERENCES marcas(marcas_id)
);
-- PASO 1: Crear nueva tabla
CREATE TABLE colores_modelo (
    color_modelo_id INT AUTO_INCREMENT PRIMARY KEY,
    modelo_id INT NOT NULL,
    color_nombre VARCHAR(30) NOT NULL,
    orden_color ENUM('principal', 'secundario', 'terciario') NOT NULL,
    FOREIGN KEY (modelo_id) REFERENCES modelos(modelos_id)
);

-- Tabla de ventas
CREATE TABLE ventas (
ventas_id INT PRIMARY KEY AUTO_INCREMENT,
fecha DATE NOT NULL,
cantidad INT,
modelos_id INT NOT NULL,
FOREIGN KEY (modelos_id) REFERENCES modelos(modelos_id)
);
-- Tabla de colores hexagesimales
CREATE TABLE color_hex (
    color_hex_id INT AUTO_INCREMENT PRIMARY KEY,
    modelos_id INT NOT NULL,

    color_1 VARCHAR(7),
    color_2 VARCHAR(7),
    color_3 VARCHAR(7),

    CONSTRAINT fk_color_hex_modelos
        FOREIGN KEY (modelos_id)
        REFERENCES modelos(modelos_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
